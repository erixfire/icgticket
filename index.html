// Log changes for history

if (ticket.status !== newStatus) {

commentText += `Status changed from ${ticket.status} to ${newStatus}. `;

ticket.status = newStatus;

updated = true;

}

if (ticket.assignedAgentUid !== newAssignedAgent) {

const oldAgentName = ticket.assignedAgentUid ? getUserName(ticket.assignedAgentUid) : 'Unassigned';

const newAgentName = newAssignedAgent ? getUserName(newAssignedAgent) : 'Unassigned';

commentText += `Assignment changed from ${oldAgentName} to ${newAgentName}.`;

ticket.assignedAgentUid = newAssignedAgent;

// If assigned, change status to 'Assigned' if it was 'New' or 'Open'

if (newAssignedAgent && (ticket.status === 'New' || ticket.status === 'Open')) {

ticket.status = 'Assigned';

}

updated = true;

}

 

if (updated) {

ticket.updatedAt = now;

 

// Add an automatic comment for the change (as internal note)

const changeComment = {

commentId: generateId('c'),

ticketId: currentTicketId,

userId: currentUser.userId, // Action performed by current user

createdAt: now,

isInternalNote: true // System updates are internal
