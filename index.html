                case 'Closed': return 'bg-red-100 text-red-800';

                default: return 'bg-gray-100 text-gray-800';

            }

        };

 

         // Get priority color class (optional, for visual flair)

        const getPriorityClass = (priority) => {

            switch (priority) {

                case 'Urgent': return 'text-red-600 font-bold';

                case 'High': return 'text-orange-600 font-semibold';

                case 'Medium': return 'text-yellow-600';

                case 'Low': return 'text-green-600';

                default: return 'text-gray-600';

            }

        };

 

 

        // --- View Management ---

        const switchView = (view) => {

            currentView = view;

            // Hide all sections/views first

            loginSection.classList.add('hidden');

            appSection.classList.add('hidden');

            dashboardView.classList.add('hidden');

            createTicketView.classList.add('hidden');

            ticketDetailView.classList.add('hidden');

 

            // Show the target section/view

            if (view === 'login') {

                loginSection.classList.remove('hidden');

                userInfo.classList.add('hidden'); // Hide user info on login screen

            } else {

                appSection.classList.remove('hidden');

                userInfo.classList.remove('hidden'); // Show user info when logged in

                if (view === 'dashboard') {

                    dashboardView.classList.remove('hidden');

                    sectionTitle.textContent = 'Dashboard';

                    createTicketButton.classList.remove('hidden'); // Show create button on dashboard

                    renderTicketList(); // Refresh list when switching to dashboard

                } else if (view === 'create') {

                    createTicketView.classList.remove('hidden');

                    sectionTitle.textContent = 'Create New Ticket';

                    createTicketButton.classList.add('hidden'); // Hide create button when creating

                    resetCreateForm();

                } else if (view === 'detail') {

                    ticketDetailView.classList.remove('hidden');

                    sectionTitle.textContent = `Ticket Details: ${currentTicketId}`;

                    createTicketButton.classList.add('hidden'); // Hide create button on detail view

                    renderTicketDetails(); // Render details when switching

                }

            }

        };

 

        // --- Rendering Functions ---

 

        // Populate dropdowns that don't change often

        const populateStaticDropdowns = () => {

            // Login dropdown

            emailSelect.innerHTML = '<option value="" disabled selected>-- Select User --</option>';

            users.forEach(user => {

                const option = document.createElement('option');

                option.value = user.userId;

                option.textContent = `${user.name} (${user.email}) - ${user.role}`;

                emailSelect.appendChild(option);

            });

 

            // Office dropdowns (Create Ticket, Filter)

            const officeOptionsHtml = offices.map(office => `<option value="${office.id}">${office.name} (${office.abbreviation})</option>`).join('');

            ticketTargetOffice.innerHTML = `<option value="" disabled selected>-- Select Target Office --</option>${officeOptionsHtml}`;

            filterOffice.innerHTML = `<option value="all">All Offices</option>${officeOptionsHtml}`;

 

             // Status dropdown for detail view (Agent Actions)

            const statusOptions = ['Open', 'Assigned', 'In Progress', 'On Hold', 'Resolved', 'Closed'];

            updateStatus.innerHTML = statusOptions.map(s => `<option value="${s}">${s}</option>`).join('');

        };

 

        // Render the list of tickets in the dashboard

        const renderTicketList = () => {

            if (!currentUser) return; // Don't render if not logged in

 

            const statusFilter = filterStatus.value;

            const officeFilter = (currentUser.role === 'admin') ? filterOffice.value : 'all'; // Only admins use office filter

 

            let filteredTickets = tickets.filter(ticket => {

                // Role-based filtering

                if (currentUser.role === 'submitter' && ticket.submitterUid !== currentUser.userId) {

                    return false; // Submitters only see their own tickets

                }

                if (currentUser.role === 'agent' && ticket.targetOfficeId !== currentUser.officeId && ticket.assignedAgentUid !== currentUser.userId) {

                     // Agents see tickets for their office OR directly assigned to them

                     return false;

                }

 

                // Status filter

                if (statusFilter !== 'all' && ticket.status !== statusFilter) {

                    return false;

                }

 

                 // Office filter (for admin)

                 if (currentUser.role === 'admin' && officeFilter !== 'all' && ticket.targetOfficeId !== officeFilter) {

                    return false;

                 }

 

                return true; // Include ticket if it passes all filters

            });

 

            // Sort by creation date descending

            filteredTickets.sort((a, b) => b.createdAt - a.createdAt);

 

            ticketList.innerHTML = ''; // Clear existing list

 

            if (filteredTickets.length === 0) {

                ticketList.innerHTML = `<tr class="h-20"><td colspan="8" class="text-center text-gray-500 py-6">No tickets found matching your criteria.</td></tr>`;

                return;

            }

 

            filteredTickets.forEach(ticket => {

                const row = document.createElement('tr');

                row.className = 'hover:bg-gray-50 cursor-pointer';

                row.dataset.ticketId = ticket.ticketId; // Store ID for click handling

 

                row.innerHTML = `

                    <td class="px-6 py-4 whitespace-nowrap text-sm font-medium text-gray-900">${ticket.ticketId}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-700 max-w-xs truncate" title="${ticket.title}">${ticket.title}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm">

                        <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${getStatusClass(ticket.status)}">

                            ${ticket.status}

                        </span>

                    </td>

                     <td class="px-6 py-4 whitespace-nowrap text-sm ${getPriorityClass(ticket.priority)}">${ticket.priority}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${getOfficeName(ticket.targetOfficeId)}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${getUserName(ticket.submitterUid)}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${formatDate(ticket.createdAt)}</td>

                    <td class="px-6 py-4 whitespace-nowrap text-right text-sm font-medium">

                        <button class="text-indigo-600 hover:text-indigo-900 view-button" data-ticket-id="${ticket.ticketId}">View</button>

                    </td>

                `;

                 // Add click listener to the whole row (or just the button)

                row.addEventListener('click', (e) => {

                     // Check if the click was on the button or the row itself

                     // if (e.target.classList.contains('view-button') || e.target.closest('tr')) {

                        currentTicketId = ticket.ticketId;

                        switchView('detail');

                    // }

                });

                ticketList.appendChild(row);

            });

             // Adjust table layout after rendering rows

             adjustTableLayout();

        };

 

         // Adjust table layout to prevent header/body misalignment with scrollbar

         const adjustTableLayout = () => {

            const table = ticketList.closest('table');

            const thead = table.querySelector('thead');

            const tbody = table.querySelector('tbody');

            if (tbody.scrollHeight > tbody.clientHeight) { // Check if scrollbar is visible

                thead.style.paddingRight = '8px'; // Match scrollbar width

            } else {

                thead.style.paddingRight = '0px';

            }

         };

 

        // Render details of the selected ticket

        const renderTicketDetails = () => {

            const ticket = tickets.find(t => t.ticketId === currentTicketId);

            if (!ticket) {

                showMessage('Error: Ticket not found.', 'error');

                switchView('dashboard');

                return;

            }

 

            // Populate basic info

            detailTicketId.textContent = ticket.ticketId;

            detailStatus.textContent = ticket.status;

            detailStatus.className = `px-2 py-0.5 rounded-full text-xs font-medium ${getStatusClass(ticket.status)}`; // Apply color

            detailPriority.textContent = ticket.priority;

            detailPriority.className = getPriorityClass(ticket.priority); // Apply color/style

            detailSubject.textContent = ticket.title;

            detailCategory.textContent = ticket.category;

            detailCreatedDate.textContent = formatDate(ticket.createdAt);

            detailSubmitter.textContent = getUserName(ticket.submitterUid);

            detailSubmitterOffice.textContent = getOfficeName(ticket.submitterOfficeId);

            detailTargetOffice.textContent = getOfficeName(ticket.targetOfficeId);

            detailAssignedAgent.textContent = ticket.assignedAgentUid ? getUserName(ticket.assignedAgentUid) : 'N/A';

            detailUpdatedDate.textContent = formatDate(ticket.updatedAt);

            detailDescription.textContent = ticket.description;

 

            // Render attachments (simulation)

            // In real app, fetch attachment list from Firestore/Storage

            detailAttachments.innerHTML = '<p class="text-gray-500 italic">Attachment simulation - No files available.</p>';

 

            // Render comments/history

            renderComments(ticket.ticketId);

 

            // Show/Hide Agent Actions and populate agent dropdown

            if (currentUser.role === 'agent' || currentUser.role === 'admin') {

                agentActions.classList.remove('hidden');

                internalNoteCheckboxContainer.classList.remove('hidden'); // Show internal note checkbox

 

                // Populate status dropdown (set current status)

                updateStatus.value = ticket.status;

 

                // Populate assign agent dropdown (only agents from the target office + admin)

                const targetOfficeAgents = users.filter(u => u.role === 'agent' && u.officeId === ticket.targetOfficeId);

                // Add admin if not already an agent in that office

                const adminUser = users.find(u => u.role === 'admin');

                let possibleAssignees = [...targetOfficeAgents];

                if (adminUser && !possibleAssignees.some(a => a.userId === adminUser.userId)) {

                    possibleAssignees.push(adminUser);

                }

 

                assignAgent.innerHTML = '<option value="">-- Unassigned --</option>';

                possibleAssignees.forEach(agent => {

                    const option = document.createElement('option');

                    option.value = agent.userId;

                    option.textContent = `${agent.name} (${getOfficeName(agent.officeId)})`;

                    option.selected = ticket.assignedAgentUid === agent.userId;

                    assignAgent.appendChild(option);

                });

 

            } else {

                agentActions.classList.add('hidden');

                 internalNoteCheckboxContainer.classList.add('hidden'); // Hide internal note checkbox

            }

 

            // Reset comment form

            addCommentForm.reset();

            internalNote.checked = false;

        };

 

        // Render comments for a specific ticket

        const renderComments = (ticketId) => {

            const ticketComments = comments

                .filter(c => c.ticketId === ticketId)

                .sort((a, b) => a.createdAt - b.createdAt); // Sort chronologically

 

            detailCommentsHistory.innerHTML = ''; // Clear existing comments

 

            if (ticketComments.length === 0) {

                detailCommentsHistory.innerHTML = '<p class="text-gray-500 italic text-sm">No comments or history yet.</p>';

                return;

            }

 

            ticketComments.forEach(comment => {

                // Hide internal notes from submitters

                if (comment.isInternalNote && currentUser.role === 'submitter') {

                    return;

                }

 

                const commentDiv = document.createElement('div');

                commentDiv.className = `p-2 rounded text-sm ${comment.isInternalNote ? 'bg-yellow-50 border border-yellow-200' : 'bg-white'}`;

 

                const commenterName = getUserName(comment.userId);

                const timestamp = formatDate(comment.createdAt);

                const internalLabel = comment.isInternalNote ? '<span class="font-semibold text-yellow-700">[Internal Note]</span>' : '';

 

                commentDiv.innerHTML = `

                    <p class="font-semibold text-gray-700">${commenterName} ${internalLabel}</p>

                    <p class="text-gray-600 whitespace-pre-wrap">${comment.commentText}</p>

                    <p class="text-xs text-gray-400 text-right">${timestamp}</p>

                `;

                detailCommentsHistory.appendChild(commentDiv);

            });

             // Scroll to the bottom of the comments

             detailCommentsHistory.scrollTop = detailCommentsHistory.scrollHeight;

        };

 

 

        // Reset the create ticket form

        const resetCreateForm = () => {

            createTicketForm.reset();

            // Reset file input visually (though JS can't fully reset it for security)

            document.getElementById('ticketAttachments').value = '';

        };

 

        // --- Event Handlers ---

 

        // Login

        loginForm.addEventListener('submit', (e) => {

            e.preventDefault();

            const selectedUserId = emailSelect.value;

            if (selectedUserId) {

                currentUser = users.find(u => u.userId === selectedUserId);

                if (currentUser) {

                    // Update user info display

                    userName.textContent = currentUser.name;

                    userOfficeName.textContent = getOfficeName(currentUser.officeId);

                    userRole.textContent = currentUser.role;

 

                    // Show/hide admin-specific elements

                    if (currentUser.role === 'admin') {

                        filterOfficeContainer.classList.remove('hidden');

                    } else {

                        filterOfficeContainer.classList.add('hidden');

                    }

 

                    showMessage(`Welcome, ${currentUser.name}!`, 'success');

                    switchView('dashboard');

                } else {

                     showMessage('Error: User not found.', 'error');

                }

            } else {

                 showMessage('Please select a user to login.', 'error');

            }

        });

 

        // Logout

        logoutButton.addEventListener('click', () => {

            currentUser = null;

            currentTicketId = null;

            switchView('login');

             showMessage('You have been logged out.', 'success');

        });

 

        // Navigate to Create Ticket view

        createTicketButton.addEventListener('click', () => {

            switchView('create');

        });

 

        // Cancel Ticket Creation

        cancelCreateTicket.addEventListener('click', () => {

            switchView('dashboard');

        });

 

        // Submit New Ticket

        createTicketForm.addEventListener('submit', (e) => {

            e.preventDefault();

            if (!currentUser) return;

 

            const formData = new FormData(createTicketForm);

            const now = new Date();

 

            const newTicket = {

                ticketId: generateId('TKT'),

                title: formData.get('subject'),

                description: formData.get('description'),

                submitterUid: currentUser.userId,

                submitterOfficeId: currentUser.officeId,

                targetOfficeId: formData.get('targetOffice'),

                assignedAgentUid: null,

                category: formData.get('category'),

                priority: formData.get('priority'),

                status: 'New', // Initial status

                createdAt: now,

                updatedAt: now,

                resolutionNotes: null

            };

 

            // Simulate adding to Firestore

            tickets.unshift(newTicket); // Add to the beginning of the array

 

            // Simulate file upload (just log file names)

            const files = formData.getAll('attachments');

            if (files && files[0].size > 0) { // Check if files were actually selected

                console.log("Simulating upload for:", files.map(f => f.name));

                // In a real app: Upload files to Firebase Storage and store references.

            }

 

 

            showMessage(`Ticket ${newTicket.ticketId} created successfully!`, 'success');

            switchView('dashboard'); // Go back to dashboard

        });

 

        // Apply Dashboard Filters

        applyFilterButton.addEventListener('click', () => {

            renderTicketList(); // Re-render list with current filter values

        });

 

        // Back to Dashboard from Detail View

        backToDashboardButton.addEventListener('click', () => {

            currentTicketId = null;

            switchView('dashboard');

        });

 

        // Update Ticket Status/Assignment (Agent Action)

        updateTicketButton.addEventListener('click', () => {

            if (!currentUser || (currentUser.role !== 'agent' && currentUser.role !== 'admin')) return;

 

            const ticket = tickets.find(t => t.ticketId === currentTicketId);

            if (!ticket) return;

 

            const newStatus = updateStatus.value;

            const newAssignedAgent = assignAgent.value || null; // Use null if unassigned

            const now = new Date();

            let updated = false;

            let commentText = '';

 

            // Log changes for history

            if (ticket.status !== newStatus) {

                 commentText += `Status changed from ${ticket.status} to ${newStatus}. `;

                 ticket.status = newStatus;

                 updated = true;

            }

            if (ticket.assignedAgentUid !== newAssignedAgent) {

                const oldAgentName = ticket.assignedAgentUid ? getUserName(ticket.assignedAgentUid) : 'Unassigned';

                const newAgentName = newAssignedAgent ? getUserName(newAssignedAgent) : 'Unassigned';

                 commentText += `Assignment changed from ${oldAgentName} to ${newAgentName}.`;

                 ticket.assignedAgentUid = newAssignedAgent;

                 // If assigned, change status to 'Assigned' if it was 'New' or 'Open'

                 if (newAssignedAgent && (ticket.status === 'New' || ticket.status === 'Open')) {

                    ticket.status = 'Assigned';

                 }

                 updated = true;

            }

 

            if (updated) {

                ticket.updatedAt = now;

 

                 // Add an automatic comment for the change (as internal note)

                 const changeComment = {

                    commentId: generateId('c'),

                    ticketId: currentTicketId,

                    userId: currentUser.userId, // Action performed by current user

                    commentText: `System Update: ${commentText.trim()}`,

                    createdAt: now,

                    isInternalNote: true // System updates are internal

                 };

                 comments.push(changeComment);

 

                showMessage('Ticket updated successfully!', 'success');

                renderTicketDetails(); // Re-render details to show changes

            } else {

                 showMessage('No changes detected.', 'info', 1500);

            }

        });

 

        // Add Comment

        addCommentForm.addEventListener('submit', (e) => {

            e.preventDefault();

            if (!currentUser || !currentTicketId) return;

 

            const commentTextValue = newComment.value.trim();

            if (!commentTextValue) {

                showMessage('Comment cannot be empty.', 'error', 1500);

                return;

            }

 

            const isInternal = (currentUser.role === 'agent' || currentUser.role === 'admin') && internalNote.checked;

 

            const newCommentObj = {

                commentId: generateId('c'),

                ticketId: currentTicketId,

                userId: currentUser.userId,

                commentText: commentTextValue,

                createdAt: new Date(),

                isInternalNote: isInternal

            };

 

            // Simulate adding to Firestore

            comments.push(newCommentObj);

 

            // Update the ticket's updatedAt timestamp

            const ticket = tickets.find(t => t.ticketId === currentTicketId);

            if(ticket) ticket.updatedAt = newCommentObj.createdAt;

 

 

            showMessage('Comment added successfully!', 'success');

            renderComments(currentTicketId); // Re-render comments section

            addCommentForm.reset(); // Clear the form

            internalNote.checked = false;

 

            // Also update the main list view if visible (though unlikely)

            if (currentView === 'dashboard') {

                renderTicketList();

            } else if (currentView === 'detail') {

                 // Update the "Last Updated" field in the detail view immediately

                 detailUpdatedDate.textContent = formatDate(newCommentObj.createdAt);

            }

 

        });

 

 

        // --- Initial Setup ---

        populateStaticDropdowns();

        switchView('login'); // Start at the login screen

 

        // Add resize listener to adjust table layout

        window.addEventListener('resize', adjustTableLayout);
